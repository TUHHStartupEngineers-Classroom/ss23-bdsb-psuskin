1.2 Set column types

col_types <- list(
  id = col_character(),
  type = col_character(),
  number = col_character(),
  country = col_character(),
  date = col_date("%Y-%m-%d"),
  abstract = col_character(),
  title = col_character(),
  kind = col_character(),
  num_claims = col_double(),
  filename = col_character(),
  withdrawn = col_double()
)

1.3 Load tables

patent_tb1 <- vroom(
  file       = '../../Patent_data_reduced/patent.tsv', 
  delim      = "\t", 
  col_types  = col_types,
  na         = c("", "NA", "NULL")
)
#> Warning: The following named parsers don't match the column names: type, number,
#> country, abstract, title, kind, filename, withdrawn
assignee_tb1 <- vroom(
  file       = '../../Patent_data_reduced/assignee.tsv', 
  delim      = "\t", 
  col_types  = col_types,
  na         = c("", "NA", "NULL")
)
#> Warning: The following named parsers don't match the column names: number,
#> country, date, abstract, title, kind, num_claims, filename, withdrawn
patent_assignee_tb1 <- vroom(
  file       = '../../Patent_data_reduced/patent_assignee.tsv', 
  delim      = "\t", 
  col_types  = col_types,
  na         = c("", "NA", "NULL"))
#> Warning: The following named parsers don't match the column names: id, type,
#> number, country, date, abstract, title, kind, num_claims, filename, withdrawn
uspc_tb1 <- vroom(
  file       = '../../Patent_data_reduced/uspc.tsv', 
  delim      = "\t", 
  col_types  = col_types,
  na         = c("", "NA", "NULL")) %>%
  transform(patent_id = as.character(patent_id))
#> Warning: The following named parsers don't match the column names: id, type,
#> number, country, date, abstract, title, kind, num_claims, filename, withdrawn

1.4 Wrangle tables

wrangled <- assignee_tb1 %>%
  left_join(patent_assignee_tb1, by = c("id" = "assignee_id")) %>%
  left_join(patent_tb1, by = c("patent_id" = "id")) %>%
  left_join(uspc_tb1, by = "patent_id")
#> Warning in left_join(., uspc_tb1, by = "patent_id"): Detected an unexpected

2 Question 1

2.1 Get patent leaders

patent_leaders <- sort(table(wrangled$organization), decreasing=T)[1:10] %>%
  as.data.frame() %>%
  mutate(Var1 = Var1 %>% str_to_title())

2.2 Display patent leaders in table

data.table(
  "Organization" = patent_leaders$Var1,
  "No. of patents (2014)" = patent_leaders$Freq)

3 Question 2

3.1 Get August wrangled table

wrangled_august <- wrangled %>% 
  select(organization, date) %>%
  filter(date >= "2014-08-01" & date <= "2014-08-31")

3.2 Get August patent leaders

patent_leaders_august <- sort(table(wrangled_august$organization), decreasing=T)[1:10] %>%
  as.data.frame() %>%
  mutate(Var1 = Var1 %>% str_to_title())

3.3 Display August patent leaders in table

data.table(
  "Organization" = patent_leaders_august$Var1,
  "No. of patents (August 2014)" = patent_leaders_august$Freq)

4 Question 3

4.1 Get class wrangled table

wrangled_class <- wrangled %>%
  select(organization, mainclass_id) %>%
  filter(organization %in% patent_leaders$Var1[1:10]) %>% 
  subset(mainclass_id != "No longer published")

4.2 Get class patent leaders

class_leaders <- sort(table(wrangled_class$mainclass_id), decreasing=T)[1:5] %>% 
  as.data.frame() %>%
  mutate(Var1 = Var1 %>% str_to_title())

4.3 Display class patent leaders in table

data.table(
  "USPTO tech main class (id since can't find name) for top 10 companies (US since can't find worldwide)" = class_leaders$Var1,
  "No. of patents (2014)" = class_leaders$Freq)
